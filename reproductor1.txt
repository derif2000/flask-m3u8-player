<body>
    <h1>Probador de Endpoint M3U8</h1>
    <input type="text" id="videoUrl" placeholder="Ingresa URL del video" style="width: 400px;">
    <button onclick="testEndpoint()">Probar</button>
    <div id="result" style="margin-top: 20px; white-space: pre-wrap;"></div>

    <script>
        function testEndpoint() {
            const url = document.getElementById('videoUrl').value;
            const resultDiv = document.getElementById('result');
            
            if (!url) {
                resultDiv.textContent = 'Por favor ingresa una URL';
                return;
            }

            resultDiv.textContent = 'Consultando...';
            
            // Usando GET
            fetch(`http://35.239.208.148:5000/get_m3u8?url=${encodeURIComponent(url)}`)
                .then(response => response.json())
                .then(data => {
                    resultDiv.textContent = JSON.stringify(data, null, 2);
                })
                .catch(error => {
                    resultDiv.textContent = 'Error: ' + error.message;
                });
        }
    </script>
